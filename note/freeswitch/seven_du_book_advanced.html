    • Session相关函数

      session:answer()
session.sleep(1000)
session:streamFile("/tmp/hello.wav")
-- 没有对应的函数，可以使用execute执行
-- session:execute("playback", "/tmp/sound.wav")

      1. lua脚本执行完，会自动挂机
        -- 设置不自动挂机
session:setAutoHangup(false)
      2. getVariable
        local moh = session:getVariable("hold_music")
local context = session:getVariable("context")
local dest_number = session:getVariable("destination_number")
local cid_name = session:getVariable("caller_id_name")
local cid_number = session:getVariable("caller_id_number")
local ani = session:getVariable("ani")
local uuid = session:getVariable("uuid")
      3. getUUID
        local uuid = session:get_uuid()
-- 等价于
local uuid = session:getVariable("uuid")
      4. setVariable。设置通道变量，等价于Dialplan App里的set
        session:setVariable("varname", "varval")
      5. hangup。挂断当前通话
        session:hangup()
-- 指定挂断原因
session:hangup("USER_BUSY")
      6. ready。检查session是否可正常使用，挂机返回false
        -- 有循环一定要检测session状态
while (session:ready() == true) do
 -- do something here
end
      7. streamFile。放音，相当于playback
        session:streamFile("/tmp/test.wav")
      8. recordFile：录音，相当于record ``` session:recordFile(file_name [,max_len_secs] [,silence_threshold], [,silence_secs]) -- file_name：文件名 -- max_len_secs：录音最长秒数 -- silence_threshold：声音阈值，小于该声音就认为是静音 -- silence_secs：如果静音时长大于一定秒数，则停止录音

      session:recordFile("/tmp/test_record.wav")

      8. read：相当于dialplan App中的read，用于播放一个声音，并获取DTMF

      digits = session:read(15, 18, "/tmp/input-id-card.wav", "5000", "#") session:consoleLog("info", "INFO ID Card Number: " .. digits .. "\n")

      9. palyAndGetDigits。相当于play_and_get_digits。参数格式是：

      , , , , , , , , [variable_name], [digit_timeout], [transfer_on_failure] -- timeout：收齐所有号的超时值 -- digit_timeout：允许两次按键之间的时间间隔最大值 -- transfer_on_failure：失败后转到dialplan的一个extension -- 如：failed XML dialplan

      digits = session:playAndGetDigits(15, 18, 3, 10000, "#", 

      "/tmp/input-id-card.wav", "/tmp/invalid_num.wav",
"^\\d{15}|\\d{17}|[0-9\\*]$")

      session:execute("log", "INFO ID Card Number: " .. digits .. "\n")

      10. setInputCallback。在放音或录音时，用户按下的DTMF可以用于触发一些功能。

      -- type：dtmf -- obj：obj.digit(链值)和obj.duration(时长) function onInputCBF(s, type, obj, arg)

      if (type == "dtmf") then
    freeswitch.consoleLog("INFO", "Got DTMF: " ..
        " Duration: " .. obj.digit .. obj.duration .. "\\n")
    if (obj.digit == "3") then
        return "break"
    end
end
return ''

      end

      session:setInputCallback('onInputCBF', ''); session:streamFile("local_stream://moh"); ```

    • 非Session函数和独立的Lua脚本

      -- 日志输出
freeswitch.consoleLog("NOTICE", "Hello lua log!\n")
-- 允许在lua执行任意的API
api = freeswitch.API()
reply = api:execute("version", "")
freeswitch.consoleLog("INFO", reply .. "\n")

-- 执行
freeswitch> lua lua/version.lua
-ERR no reply

freeswitch> luarun lua/version.lua
+OK

-- 其他
freeswitch.bridge()
freeswitch.email()

      lua是一个App，自动获得一个session对象
      luarun是一个API，可在控制台独立运行。不会自动获得session，但可以自己创建
